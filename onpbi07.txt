	/**
	 * 이동 화면으로부터 받은 DATA SET
	 */
	scwin.getData = function(retdata) {
		try {
			var objmsg;
            
            if (typeof retdata == 'undefined') {
                scvar.param = com.getParameter();
                objmsg = scvar.param;
            } else {
                objmsg = retdata;
            }
			
			var objChange;
			if(com.empty(objmsg.tran)){
				objChange = objmsg.id
			}else{
				objChange = objmsg.tran
			}
			
			switch(objChange){
			
				// 물건리스트조회 팝업
				case "onpmp02":

					scvar.SPC_CODE 	= objmsg.gSPC_CODE; //spc코드
					scvar.DEBT_NO 	= objmsg.gDEBT_NO;  //차주번호
					scvar.OBJT_NO 	= objmsg.gOBJT_NO;  //물건번호
					//scvar.BOND_SEQ 	= scvar.SPC_CODE + scvar.DEBT_NO + scvar.OBJT_NO; // 채권일련번호
					scvar.STAF_ID 	= objmsg.gSTAF_ID;	//담당자
					scvar.DEBT_NAME = objmsg.gDEBT_NAME; //차주명
					scvar.MORT_KIND  = objmsg.gMORT_KIND; //담보종류
					
					var tmpDebtNo =objmsg.gDEBT_NO.split("-");
					
					sbx_keyDebtNoCode.setText(tmpDebtNo[0]);
					ibx_keyDebtNo.setValue(tmpDebtNo[1]);
                    ibx_keyDebtName.setValue(scvar.DEBT_NAME);
					
					scwin.mainProcess('onpmc01_s02'); 	//담보정보
					scwin.mainProcess('onpmc01_s09'); 	//물건사진리스트
					
					break;
				
				// 사원명 					
				case "ocmab01":
				
					ibx_keyStafId.setValue( objmsg.gSTAF_CODE );
					ibx_keyStafName.setValue( objmsg.gSTAF_NAME );
					
					break;
					
		        default:
		            break;
	        }
		} catch (ex) {
			$p.log("[scwin.getData] Exception :: " + ex);
		}
	};

		try{

			var dltList= returnData.dlt_rsList[0];

			var tmp_message = returnData.rsMsg.message;
			
			if (tmp_statusCode.substring(0,1) == "0") {
				var occCnt		= returnData.RCXXCNTF - 0;
				var tmpDatas 	= returnData.dlt_rsList;
				
				if(occCnt == 1)	{
